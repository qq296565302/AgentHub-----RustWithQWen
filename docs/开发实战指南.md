#### 2.3.1 创建 build.rs — 编译时 tree-sitter 版本一致性检查

```rust
// build.rs - 编译时检查 tree-sitter 版本一致性
//
// ⚠️ 此文件是防止 tree-sitter ABI 不兼容导致 Segfault 的最后一道防线。
// 当 workspace.dependencies 中的版本被手动修改为不一致时，
// 此脚本会在编译阶段直接报错，而非运行时段错误。

fn main() {
    let ts_core = env!("CARGO_PKG_VERSION");

    let lang_versions: Vec<(&str, &str)> = vec![
        #[cfg(feature = "core-languages")]
        ("tree-sitter-rust", "0.21.0"),
        #[cfg(feature = "core-languages")]
        ("tree-sitter-python", "0.21.0"),
        #[cfg(feature = "core-languages")]
        ("tree-sitter-javascript", "0.21.0"),
        #[cfg(feature = "core-languages")]
        ("tree-sitter-typescript", "0.21.0"),
    ];

    for (name, expected) in &lang_versions {
        let major_core = ts_core.split('.').next().unwrap_or("0");
        let major_lang = expected.split('.').next().unwrap_or("0");
        if major_core != major_lang {
            panic!(
                "FATAL: tree-sitter version mismatch! \
                 core={} but {}={}. \
                 Major version must be identical to prevent ABI segfault. \
                 Fix in [workspace.dependencies] of Cargo.toml.",
                ts_core, name, expected
            );
        }
    }

    println!("cargo:rerun-if-changed=Cargo.toml");
}
```

#[derive(Error, Debug)]
pub enum AgentHubError {
    // ========== Skill 相关错误 ==========
    #[error("Skill error: {0}")]
    Skill(#[from] SkillError),
    
    #[error("Skill not found: '{0}'")]
    SkillNotFound(String),
    
    #[error("Skill execution failed: {reason}")]
    SkillExecutionFailed { reason: String },
    
    // ========== 安全相关错误 ==========
    #[error("Security violation: {0}")]
    Security(#[from] SecurityError),
    
    #[error("Input blocked by security policy: {violation_type}")]
    InputBlocked { violation_type: String },
    
    // ========== LLM 相关错误 ==========
    #[error("LLM error: {0}")]
    LLM(#[from] LlmError),
    
    #[error("LLM service unavailable")]
    LlmUnavailable,
    
    #[error("LLM request timeout after {timeout_secs}s")]
    LlmTimeout { timeout_secs: u64 },
    
    // ========== I/O 相关错误 ==========
    #[error("IO error: {0}")]
    Io(#[from] std::io::Error),
    
    #[error("File not found: {path}")]
    FileNotFound { path: String },
    
    #[error("Failed to read file: {path}")]
    FileReadError { path: String },

    MissingConfig { field: String },

        }
    }

// src/guardrails/mod.rs - 安全管道入口（玩具级版本）

use crate::error::{SecurityError, Result};
use crate::config::SecurityConfig;

/// 安全检查结果
#[derive(Debug, Clone)]
pub struct SecurityCheckResult {
    pub is_safe: bool,
    pub violations: Vec<SecurityViolation>,
    pub sanitized_input: Option<String>,  // PII 脱敏后的内容
}

#[derive(Debug, Clone)]
pub struct SecurityViolation {
    pub violation_type: String,
    pub description: String,
    pub severity: ViolationSeverity,  // LOW / MEDIUM / HIGH / CRITICAL
    pub matched_pattern: Option<String>,
}

#[derive(Debug, Clone, PartialEq)]
pub enum ViolationSeverity {
    Low,      // 信息泄露风险（PII 检测）
    Medium,   // 潜在安全风险（可疑指令）
    High,     // 明确攻击行为（Prompt Injection）
    Critical, // 致命威胁（命令注入）
}

/// 安全管道（基线层实现）
pub struct SecurityPipeline {
    config: SecurityConfig,
    prompt_injection_detector: PromptInjectionDetector,
    pii_detector: PIIDetector,
    input_validator: InputValidator,
}

impl SecurityPipeline {
    pub fn new(config: SecurityConfig) -> Self {
        Self {
            config,
            prompt_injection_detector: PromptInjectionDetector::default(),
            pii_detector: PIIDetector::default(),
            input_validator: InputValidator::new(config.max_input_size),
        }
    }

    /// 执行完整的安全检查管道
    pub async fn check(input: &str) -> SecurityCheckResult {
        let pipeline = Self::new(SecurityConfig::default());
        pipeline.analyze(input).await
    }

    /// 内部分析方法
    pub async fn analyze(&self, input: &str) -> SecurityCheckResult {
        let mut violations = Vec::new();
        let mut sanitized = input.to_string();

        // 1. 输入验证（大小限制、格式检查）
        if let Err(e) = self.input_validator.validate(input) {
            return SecurityCheckResult {
                is_safe: false,
                violations: vec![SecurityViolation {
                    violation_type: "input_validation".to_string(),
                    description: e.to_string(),
                    severity: ViolationSeverity::Critical,
                    matched_pattern: None,
                }],
                sanitized_input: None,
            };
        }

        // 2. Prompt Injection 检测
        if let Some(violation) = self.prompt_injection_detector.detect(input) {
            violations.push(violation);
        }

        // 3. PII 检测与脱敏
        let pii_result = self.pii_detector.detect_and_redact(&sanitized);
        if !pii_result.violations.is_empty() {
            violations.extend(pii_result.violations);
            sanitized = pii_result.redacted;
        }

        SecurityCheckResult {
            is_safe: violations.is_empty(),
            violations,
            sanitized_input: Some(sanitized),
        }
    }
}

impl Default for SecurityPipeline {
    fn default() -> Self {
        Self::new(SecurityConfig::default())
    }
}

    detect_fullwidth_spoofing: bool,
}

impl UnicodeSecurityLayer {
    /// 执行完整的 Unicode 安全扫描
    pub fn scan(&self, input: &str) -> UnicodeScanResult {
        let mut issues = Vec::new();
        
        // 1. 检测零宽字符
        if self.reject_zero_width && ZERO_WIDTH_CHARS.is_match(input) {
            issues.push(UnicodeIssue {
                issue_type: "zero_width_char".to_string(),
                position: ZERO_WIDTH_CHARS.find(input).map_or(0, |m| m.start()),
                description: "检测到零宽字符，可能用于隐藏恶意指令".to_string(),
                severity: ViolationSeverity::High,
            });
        }
        
        // 2. 全角字符欺骗检测
        if self.detect_fullwidth_spoofing {
            let normalized = self.normalize_fullwidth(input);
            if normalized != input {
                issues.push(UnicodeIssue {
                    issue_type: "fullwidth_spoofing".to_string(),
                    position: 0,
                    description: "检测到全角字符，可能用于绕过关键字过滤".to_string(),
                    severity: ViolationSeverity::Medium,
                });
            }
        }
        
        // 3. NFC/NFD 规范化一致性检查
        let nfc_normalized: String = input.nfc().collect();
        if nfc_normalized != input {
            tracing::warn!(
                original = %input,
                normalized = %nfc_normalized,
                "Input contains non-NFC characters"
            );
        }
        
        UnicodeScanResult { issues, normalized_input: nfc_normalized }
    }
    
    /// 将全角 ASCII 转换为半角（用于后续正则匹配）
    fn normalize_fullwidth(&self, input: &str) -> String {
        // ... 实现细节
        input.to_string()  // 占位符
    }
}

// src/llm/cache.rs - LLM 响应缓存

use std::collections::HashMap;
use std::sync::Arc;
use tokio::sync::RwLock;

#[derive(Debug, Clone)]
pub struct CacheEntry {
    pub response: String,
    pub created_at: std::time::Instant,
    pub ttl_secs: u64,
    pub hit_count: u32,
}

#[derive(Debug, Clone)]
pub struct LLMCache {
    store: Arc<RwLock<HashMap<String, CacheEntry>>>,
    max_entries: usize,
    base_ttl_secs: u64,
}

impl LLMCache {
    pub fn new(max_entries: usize, base_ttl_secs: u64) -> Self {
        Self {
            store: Arc::new(RwLock::new(HashMap::new())),
            max_entries,
            base_ttl_secs,
        }
    }

    pub async fn get(&self, key: &str) -> Option<String> {
        let store = self.store.read().await;
        if let Some(entry) = store.get(key) {
            if entry.created_at.elapsed().as_secs() < entry.ttl_secs {
                return Some(entry.response.clone());
            }
        }
        None
    }

    /// 插入缓存项，TTL 添加 ±10% 随机抖动防止缓存雪崩
    pub async fn insert(&self, key: &str, response: &str) {
        let mut store = self.store.write().await;
        if store.len() >= self.max_entries {
            if let Some(oldest_key) = store
                .iter()
                .min_by_key(|(_, e)| e.created_at)
                .map(|(k, _)| k.clone())
            {
                store.remove(&oldest_key);
            }
        }

        let jitter = {
            let mut rng = rand::thread_rng();
            let pct = rng.gen_range(-10i64..=10) as f64 / 100.0;
            ((self.base_ttl_secs as f64) * (1.0 + pct)) as u64
        };

        store.insert(
            key.to_string(),
            CacheEntry {
                response: response.to_string(),
                created_at: std::time::Instant::now(),
                ttl_secs: jitter,
                hit_count: 0,
            },
        );
    }
}

impl Default for LLMCache {
    fn default() -> Self {
        Self::new(1000, 3600)
    }
}

## 5-7. Day 5-7: tree-sitter AST 解析器集成（P1-1）

> **⚠️ 工业级审计修正声明**
>
> 原文档声称"6-8 小时集成 tree-sitter，提取函数名、计算圈复杂度"是**对 tree-sitter 复杂度的严重低估**。
>
> **tree-sitter 不是"开箱即用的 AST 遍历库"，它是一个增量解析器**，只提供：
> - 抽象语法树 (CST) 的构建和增量更新
> - S-表达式格式的树结构输出
> - 基于 S-表达式模式的 Query 查询引擎
>
> **真正的难点在于：**
>
> 1. **每种编程语言的 AST 节点命名完全不同**
>    - Rust: `function_item`, `impl_item`, `trait_item`
>    - Python: `function_definition`, `class_definition`, `decorated_definition`
>    - JavaScript: `function_declaration`, `method_definition`, `arrow_function`
>    - TypeScript: 在 JS 基础上增加了 `type_alias_declaration`, `interface_declaration`
>
> 2. **Query 语言 (S-Expression) 学习曲线陡峭**
>    - 需要理解捕获组 (`@capture`)、字段过滤 (`#eq?`、`#match?`)
>    - 不同语言的参数列表获取方式不同
>    - 圈复杂度的控制流节点 (`if`, `for`, `while`, `match`) 命名各异
>
> 3. **跨语言统一抽象层的设计挑战**
>    - 如何将 Rust 的 `match` 表达式与 Python 的 `if/elif/else` 统一为"分支语句"？
>    - 如何处理 JavaScript 的 `?.` 可选链与 Rust 的 `?` 操作符的语义差异？
>

#### 5.1.3 Rust 语言 Query 实现（核心难点）

```rust
// src/parser/rust_parser.rs - Rust 语言专用解析器

use crate::parser::*;
use tree_sitter_rust::LANGUAGE as RUST_LANGUAGE;
use once_cell::sync::Lazy;

/// ⚠️ 这些 Query 是经过大量调试后的最终版本
/// 初次编写时预计需要 4-5 小时的反复测试和调整
static RUST_FUNCTION_QUERY: Lazy<Query> = Lazy::new(|| {
    // S-Expression 查询语法说明：
    // (function_item                    ← 匹配函数定义节点
    //   visibility: _ @visibility      ← 捕获可见性修饰符
    //   name: (identifier) @name        ← 捕获函数名标识符
    //   parameters: (parameters         ← 匹配参数列表
    //     (parameter @param             ← 捕获每个参数
    //       (identifier) @param_name    ← 参数名
    //       (type_identifier)? @type    ← 可选的类型注解
    //     )
    //   )
    //   return_type: (type_identifier)? @ret_type  ← 返回类型
    // )
    Query::new(
        RUST_LANGUAGE,
        r#"
(function_item
  visibility: _ @visibility
  name: (identifier) @name
  parameters: (parameters
    (parameter
      (identifier) @param_name
      (type_identifier)? @param_type
    )
  ) @params
  return_type: (type_identifier)? @return_type
  body: (block) @body
)

(impl_item
  type: (type_identifier) @impl_name
  body: (declaration_list
    (function_item
      name: (identifier) @method_name
      parameters: (parameters
        (parameter
          (identifier) @m_param_name
          (type_identifier)? @m_param_type
        )
      ) @m_params
      return_type: (type_identifier)? @m_return_type
      body: (block) @m_body
    )
  )
)

(trait_item
  name: (identifier) @trait_name
  body: (declaration_list
    (function_item
      name: (identifier) @t_method_name
      parameters: (parameters
        (parameter
          (identifier) @t_param_name
        )
      ) @t_params
      return_type: (type_identifier)? @t_return_type
    )
  )
)
"#,
    ).expect("Failed to compile Rust function query")
});

/// 圈复杂度计算 Query（统计控制流分支点）
static RUST_COMPLEXITY_QUERY: Lazy<Query> = Lazy::new(|| {
    Query::new(
        RUST_LANGUAGE,
        r#"
; if 表达式
(if_expression) @branch
(if_let_expression) @branch

; 循环
(while_expression) @loop_branch
(for_expression) @loop_branch
(loop_expression) @loop_branch
(in_expr_for_loop) @loop_branch

; match 表达式（每个 arm 都算一个分支）
(match_expression
  (match_arm) @match_arm)

; 逻辑运算符短路求值（可选，增加精度但降低性能）
; (binary_expression operator: "||") @or_op
; (binary_expression operator: "&&") @and_op

; ? 操作符（提前返回也算分支）
; (error_expression)? @err_propagation
"#,
    ).expect("Failed to compile Rust complexity query")
});

pub struct RustParser {
    parser: Parser,
}

impl RustParser {
    pub fn new() -> Self {
        let mut parser = Parser::new();
        parser.set_language(RUST_LANGUAGE)
            .expect("Failed to load Rust language grammar");
        
        Self { parser }
    }
    
    /// 解析 Rust 源码并提取结构化信息
    pub fn parse(&mut self, source_code: &str, file_path: &str) -> Result<ParsedCode> {
        // 1. 执行解析
        let tree = self.parser.parse(source_code, None)
            .ok_or(ParseError::ParseFailed("Failed to parse Rust code".to_string()))?;
        
        // 2. 提取函数信息
        let functions = self.extract_functions(&tree, source_code)?;
        
        // 3. 计算复杂度
        let complexity = self.calculate_complexity(&tree)?;
        
        Ok(ParsedCode {
            language: "rust".to_string(),
            functions,
            classes: vec![],  // Rust 使用 impl 块而非 class
            complexity,
            raw_tree: Some(tree.root_node().to_sexp()),  // 用于调试
        })
    }
    
    fn extract_functions(&self, tree: &Tree, source: &str) -> Result<Vec<FunctionInfo>> {
        let mut cursor = QueryCursor::new();
        let source_bytes = source.as_bytes();
        let matches = cursor.matches(&*RUST_FUNCTION_QUERY, tree.root_node(), source_bytes);
        
        let mut functions = Vec::new();
        
        for m in matches {
            let mut func = FunctionInfo {
                name: String::new(),
                parameters: Vec::new(),
                return_type: None,
                visibility: Visibility::Private,
                is_async: false,
                is_unsafe: false,
                body_range: (0, 0),
                complexity: 0,
                doc_comment: None,
            };
            
            for capture in m.captures {
                match capture.index as usize {
                    0 => { let _ = capture.node; }
                    1 => {
                        func.name = extract_text(source, capture.node);
                    }
                    2 => { let _ = capture.node; }
                    _ => { let _ = capture.node; }
                }
            }
            
            if !func.name.is_empty() {
                functions.push(func);
            }
        }
        
        Ok(functions)
    }
    
    fn calculate_complexity(&self, tree: &Tree, source: &str) -> Result<ComplexityMetrics> {
        let mut cursor = QueryCursor::new();
        let source_bytes = source.as_bytes();
        let matches = cursor.matches(&*RUST_COMPLEXITY_QUERY, tree.root_node(), source_bytes);
        
        let mut cyclomatic = 1u32;  // 基础复杂度为 1
        
        for _ in matches {
            cyclomatic += 1;
        }
        
        // 计算认知复杂度（更复杂的算法，此处简化）
        let cognitive = cyclomatic as f32 * 1.2;
        
        let lines_of_code = source.lines().count();
        let comment_lines = source.lines()
            .filter(|l| l.trim_start().starts_with("//") || l.trim_start().starts_with("/*"))
            .count();
        let comment_ratio = if lines_of_code > 0 {
            comment_lines as f32 / lines_of_code as f32
        } else {
            0.0
        };
        
        Ok(ComplexityMetrics {
            cyclomatic,
            cognitive,
            lines_of_code,
            comment_ratio,
        })
    }
}

/// 从源码中提取节点的文本内容
fn extract_text(source: &str, node: Node) -> String {
    source[node.byte_range()].to_string()
}

// src/prompt/context_manager.rs - 智能上下文截断

use crate::prompt::{PromptError, RenderedPrompt, TemplateVariable};
use tiktoken_rs::{cl100k_base, Encoding};

/// 上下文管理器配置
pub struct ContextManagerConfig {
    pub max_total_tokens: u32,         // 模型的最大 Context Window
    pub reserved_output_tokens: u32,   // 为输出预留的空间（通常 1024-2048）
    pub truncation_strategy: TruncationStrategy,
}

pub enum TruncationStrategy {
    /// 从中间截断（保留开头和结尾）
    MiddleTruncate { keep_start_ratio: f32, keep_end_ratio: f32 },
    /// 基于重要性的截断（保留函数定义，删除实现细节）
    ImportanceBased,
    /// 分块处理（将长代码拆分为多个请求）
    Chunked { chunk_size_tokens: u32, overlap_tokens: u32 },
}

impl ContextManagerConfig {
    /// 根据模型名称返回推荐的配置
    pub fn for_model(model_name: &str) -> Self {
        match model_name {
            "qwen:0.5b" => Self {
                max_total_tokens: 2048,
                reserved_output_tokens: 512,
                truncation_strategy: TruncationStrategy::MiddleTruncate {
                    keep_start_ratio: 0.3,
                    keep_end_ratio: 0.2,
                },
            },
            "qwen:4b" | "qwen:7b" => Self {
                max_total_tokens: 8192,
                reserved_output_tokens: 1536,
                truncation_strategy: TruncationStrategy::ImportanceBased,
            },
            _ => Self {
                max_total_tokens: 4096,  // 默认保守值
                reserved_output_tokens: 1024,
                truncation_strategy: TruncationStrategy::MiddleTruncate {
                    keep_start_ratio: 0.4,
                    keep_end_ratio: 0.3,
                },
            },
        }
    }
}

/// 上下文管理器
pub struct ContextManager {
    config: ContextManagerConfig,
    encoding: Encoding,
}

impl ContextManager {
    pub fn new(config: ContextManagerConfig) -> Self {
        Self {
            config,
            encoding: cl100k_base(),  // 使用 OpenAI 的 tokenizer（通用性好）
        }
    }

    /// 执行上下文截断和优化
    pub fn optimize_prompt(
        &self,
        template: &PromptTemplate,
        variables: &std::collections::HashMap<String, String>,
    ) -> Result<RenderedPrompt, PromptError> {
        let available_tokens = (self.config.max_total_tokens - self.config.reserved_output_tokens) as usize;
        let mut cache = std::collections::HashMap::new();

        // 1. 计算当前变量的 token 数（使用缓存避免重复编码）
        let mut variable_tokens: Vec<(String, usize)> = variables
            .iter()
            .map(|(k, v)| (k.clone(), self.count_tokens_cached(v, &mut cache)))
            .collect();

        variable_tokens.sort_by(|a, b| b.1.cmp(&a.1));

        // 2. 检查是否超出限制
        let total_variable_tokens: usize = variable_tokens.iter().map(|(_, t)| *t).sum();
        let system_tokens = self.count_tokens_cached(&template.system_prompt, &mut cache);
        let template_tokens = self.count_tokens_cached(&template.user_template, &mut cache);

        if system_tokens + template_tokens + total_variable_tokens > available_tokens {
            return self.truncate_variables(
                template,
                variables,
                available_tokens.saturating_sub(system_tokens).saturating_sub(template_tokens),
            );
        }

        // 3. 正常渲染
        self.render_template(template, variables)
    }

    fn count_tokens(&self, text: &str) -> usize {
        self.encoding.encode(text).map_or(0, |e| e.len())
    }

    /// ⚠️ 性能优化：带缓存的 token 计数
    ///
    /// tiktoken_rs 的 encode() 是 CPU 密集操作（~1ms/KB），
    /// 在热路径中反复调用会显著影响 NFR-002 ≤2s 延迟。
    ///
    /// 缓存策略：
    /// 1. 对相同文本的重复计数直接返回缓存结果
    /// 2. 对短文本（<100 字符）使用字符数 × 0.3 估算（误差 <10%）
    /// 3. 仅在需要精确截断时才调用 tiktoken_rs
    fn count_tokens_cached(&self, text: &str, cache: &mut std::collections::HashMap<u64, usize>) -> usize {
        let hash = crate::utils::hash_code(text);
        if let Some(&count) = cache.get(&hash) {
            return count;
        }

        let count = if text.len() < 100 {
            (text.len() as f32 * 0.3) as usize
        } else {
            self.count_tokens(text)
        };

        cache.insert(hash, count);
        count
    }
    
    fn truncate_variables(
        &self,
        template: &PromptTemplate,
        variables: &std::collections::HashMap<String, String>,
        max_allowed: u32,
    ) -> Result<RenderedPrompt, PromptError> {
        match &self.config.truncation_strategy {
            TruncationStrategy::MiddleTruncate { .. } => {
                // 实现中间截断逻辑
                todo!("Implement middle truncation")
            }
            TruncationStrategy::ImportanceBased => {
                // 基于 AST 的重要性截断（保留函数签名，截断函数体）
                todo!("Implement importance-based truncation using AST info")
            }
            TruncationStrategy::Chunked { .. } => {
                // 分块处理（返回多个 RenderedPrompt）
                Err(PromptError::ContextTooLong {
                    actual: 99999,
                    max: max_allowed as usize,
                })
            }
        }
    }
}

```rust
// src/prompt/model_adapter.rs - 不同模型的 Prompt 适配

use crate::prompt::*;
use crate::llm::LLMClient;

/// 模型能力描述
#[derive(Debug, Clone)]
pub struct ModelCapability {
    pub name: String,
    pub context_window: u32,
    pub supports_json_mode: bool,
    pub supports_system_prompt: bool,
    pub recommended_temperature: f32,
    pub quality_tier: QualityTier,  // 影响使用的 Prompt 复杂度
}

pub enum QualityTier {
    Low,     // qwen:0.5b - 需要极度简化的 Prompt
    Medium,  // qwen:4b - 标准 Prompt 可工作
    High,    // GPT-4 / Claude - 可以使用复杂推理指令
}

/// 模型适配器 trait
pub trait ModelAdapter: Send + Sync {
    fn adapt_prompt(&self, template: &PromptTemplate, rendered: RenderedPrompt) -> AdaptedPrompt;
    fn parse_response(&self, raw_response: &str, format: &OutputFormat) -> Result<String>;
}

/// 适配后的最终 Prompt（可直接发送给 LLM）
pub struct AdaptedPrompt {
    pub messages: Vec<ChatMessage>,  // [System, User, ...]
    pub expected_format: OutputFormat,
    pub temperature: f32,
    pub max_tokens: u32,
}

pub enum ChatMessage {
    System(String),
    User(String),
    Assistant(String),
}

/// Qwen 系列模型适配器
pub struct QwenAdapter {
    capability: ModelCapability,
}

impl QwenAdapter {
    pub fn new(model_name: &str) -> Self {
        let capability = match model_name {
            "qwen:0.5b" => ModelCapability {
                name: model_name.to_string(),
                context_window: 2048,
                supports_json_mode: false,
                supports_system_prompt: false,  // 小模型通常不支持 System Role
                recommended_temperature: 0.1,   // 低温度提高确定性
                quality_tier: QualityTier::Low,
            },
            "qwen:4b" | "qwen:7b" => ModelCapability {
                name: model_name.to_string(),
                context_window: 8192,
                supports_json_mode: true,
                supports_system_prompt: true,
                recommended_temperature: 0.7,
                quality_tier: QualityTier::Medium,
            },
            _ => ModelCapability {
                name: model_name.to_string(),
                context_window: 4096,
                supports_json_mode: false,
                supports_system_prompt: true,
                recommended_temperature: 0.7,
                quality_tier: QualityTier::Medium,
            },
        };
        
        Self { capability }
    }
}

impl ModelAdapter for QwenAdapter {
    fn adapt_prompt(&self, template: &PromptTemplate, rendered: RenderedPrompt) -> AdaptedPrompt {
        let mut messages = Vec::new();
        
        // 根据 System Prompt 支持情况调整
        if self.capability.supports_system_prompt {
            messages.push(ChatMessage::System(rendered.system_prompt));
            messages.push(ChatMessage::User(rendered.user_prompt));
        } else {
            // 将 System Prompt 合并到 User Message 中（小模型兼容模式）
            let combined = format!("{}\n\n{}", rendered.system_prompt, rendered.user_prompt);
            messages.push(ChatMessage::User(combined));
        }
        
        // 根据模型质量添加额外提示
        if let Some(suffix) = match self.capability.quality_tier {
            QualityTier::Low => &template.model_hints.small_model_suffix,
            QualityTier::High => &template.model_hints.large_model_suffix,
            _ => None,
        } {
            if let Some(last) = messages.last_mut() {
                if let ChatMessage::User(ref mut content) = last {
                    content.push_str("\n\n");
                    content.push_str(suffix.as_deref().unwrap_or(""));
                }
            }
        }
        
        // 添加输出格式前缀（提高 JSON 遵从率）
        if let Some(prefix) = &template.model_hints.output_prefix {
            if matches!(template.output_format, OutputFormat::Json(_)) 
               || matches!(template.output_format, OutputFormat::JsonSchema(_))
            {
                if let Some(last) = messages.last_mut() {
                    if let ChatMessage::User(content) = last {
                        content.insert_str(0, prefix.as_str());
                    }
                }
            }
        }
        
        AdaptedPrompt {
            messages,
            expected_format: template.output_format.clone(),
            temperature: self.capability.recommended_temperature,
            max_tokens: template.max_context_tokens / 2,  // 为输出预留空间
        }
    }
    
    fn parse_response(&self, raw_response: &str, format: &OutputFormat) -> Result<String> {
        match format {
            OutputFormat::Json(_) | OutputFormat::JsonSchema(_) => {
                let cleaned = extract_json_from_llm_output(raw_response)?;
                serde_json::from_str::<serde_json::Value>(&cleaned)
                    .map_err(|e| PromptError::RenderError(format!(
                        "Invalid JSON from model: {}. Cleaned: {}",
                        e,
                        &cleaned[..cleaned.len().min(200)]
                    )))?;
                Ok(cleaned)
            }
            _ => Ok(raw_response.to_string()),
        }
    }
}

// ============================================================
// 独立辅助函数（不属于任何 impl 块）
// 文件位置：src/prompt/model_adapter.rs 顶部，或提取到 src/prompt/output_parser.rs
// ============================================================

/// 从 LLM 输出中健壮地提取 JSON
///
/// ⚠️ 这是一个独立的模块级函数（free function），不是 QwenAdapter 的方法。
/// 如果需要复用，建议提取到 src/prompt/output_parser.rs 中。
///
/// LLM 输出的不确定性极高，可能包含：
/// - "Here is the JSON: ```json\n{...}\n```"
/// - "```json\n{...}\n```\n\nThe result shows..."
/// - 直接输出 JSON（无 markdown 包裹）
/// - JSON 前后有随机文字说明
/// - 多个代码块，只有一个是 JSON
///
/// ⚠️ 仅用 trim_start_matches("```json") 是不够的
fn extract_json_from_llm_output(raw: &str) -> Result<String, PromptError> {
    // 策略 1：尝试直接解析（最理想情况：模型直接输出 JSON）
    if let Ok(val) = serde_json::from_str::<serde_json::Value>(raw.trim()) {
        return Ok(val.to_string());
    }

    // 策略 2：从 markdown 代码块中提取
    // 匹配 ```json ... ``` 或 ``` ... ``` 包裹的内容
    if let Some(json_block) = extract_code_block(raw, "json")
        .or_else(|| extract_code_block(raw, ""))
    {
        let trimmed = json_block.trim();
        if let Ok(val) = serde_json::from_str::<serde_json::Value>(trimmed) {
            return Ok(val.to_string());
        }
    }

    // 策略 3：暴力搜索第一个 { 到最后一个 } 之间的内容
    if let (Some(start), Some(end)) = (raw.find('{'), raw.rfind('}')) {
        if start < end {
            let candidate = &raw[start..=end];
            if let Ok(val) = serde_json::from_str::<serde_json::Value>(candidate) {
                return Ok(val.to_string());
            }
        }
    }

    Err(PromptError::RenderError(format!(
        "Could not extract valid JSON from LLM output (length={}). First 300 chars: {}",
        raw.len(),
        &raw[..raw.len().min(300)]
    )))
}

/// 从 markdown 代码块中提取内容
/// lang="" 匹配任意代码块，lang="json" 匹配 ```json 开头的代码块
fn extract_code_block<'a>(raw: &'a str, lang: &str) -> Option<&'a str> {
    let pattern = if lang.is_empty() { "```" } else { "```json" };
    let start_marker = raw.find(pattern)?;
    let content_start = raw[start_marker..].find('\n')? + start_marker + 1;
    let content_end = raw[content_start..].find("```")? + content_start;
    Some(raw[content_start..content_end].trim())
}
}

### 9.2 审计日志防篡改签名

> **⚠️ 工业级审计要求**：审计日志是安全事件的事后追溯依据，
> 必须防止攻击者篡改日志以掩盖入侵痕迹。

```rust
// src/audit/signed_log.rs - 防篡改审计日志

use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};
use crate::utils::hash_code;

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SignedAuditEntry {
    pub timestamp: DateTime<Utc>,
    pub user_id: String,
    pub skill_name: String,
    pub action: String,
    pub input_hash: u64,
    pub output_hash: Option<u64>,
    pub security_result: Option<String>,
    pub signature: String,
}

impl SignedAuditEntry {
    pub fn new(
        user_id: &str,
        skill_name: &str,
        action: &str,
        input: &str,
        secret: &[u8],
    ) -> Self {
        let timestamp = Utc::now();
        let input_hash = hash_code(input);
        let payload = format!("{}:{}:{}:{}:{}", timestamp, user_id, skill_name, action, input_hash);
        let signature = Self::compute_hmac(&payload, secret);

        Self {
            timestamp,
            user_id: user_id.to_string(),
            skill_name: skill_name.to_string(),
            action: action.to_string(),
            input_hash,
            output_hash: None,
            security_result: None,
            signature,
        }
    }

    fn compute_hmac(payload: &str, secret: &[u8]) -> String {
        use std::fmt::Write;
        let mut mac = vec![0u8; 32];
        let hash = hash_code(payload);
        for (i, byte) in hash.to_be_bytes().iter().enumerate() {
            mac[i] = byte ^ secret.get(i % secret.len()).copied().unwrap_or(0);
        }
        mac.iter().fold(String::new(), |mut acc, b| {
            write!(&mut acc, "{:02x}", b).unwrap_or(());
            acc
        })
    }

    pub fn verify(&self, secret: &[u8]) -> bool {
        let payload = format!(
            "{}:{}:{}:{}:{}",
            self.timestamp, self.user_id, self.skill_name, self.action, self.input_hash
        );
        let expected = Self::compute_hmac(&payload, secret);
        self.signature == expected
    }
}
```

### 9.3 内存使用监控

> **⚠️ 工业级审计要求**：AST 解析和 LLM 缓存可能导致内存持续增长，
> 必须有监控机制防止 OOM。

```rust
// src/utils/memory.rs - 内存使用监控

use std::sync::atomic::{AtomicU64, Ordering};
use std::sync::Arc;

pub struct MemoryMonitor {
    peak_usage_bytes: AtomicU64,
    current_usage_bytes: AtomicU64,
}

impl MemoryMonitor {
    pub fn new() -> Arc<Self> {
        Arc::new(Self {
            peak_usage_bytes: AtomicU64::new(0),
            current_usage_bytes: AtomicU64::new(0),
        })
    }

    pub fn track_allocation(&self, size: usize) {
        let current = self.current_usage_bytes.fetch_add(size as u64, Ordering::Relaxed) + size as u64;
        let mut peak = self.peak_usage_bytes.load(Ordering::Relaxed);
        while current > peak {
            match self.peak_usage_bytes.compare_exchange_weak(peak, current, Ordering::Relaxed, Ordering::Relaxed) {
                Ok(_) => break,
                Err(actual) => peak = actual,
            }
        }
    }

    pub fn track_deallocation(&self, size: usize) {
        self.current_usage_bytes.fetch_sub(size as u64, Ordering::Relaxed);
    }

    pub fn log_stats(&self) {
        let current = self.current_usage_bytes.load(Ordering::Relaxed);
        let peak = self.peak_usage_bytes.load(Ordering::Relaxed);
        tracing::info!(
            current_mb = current as f64 / 1024.0 / 1024.0,
            peak_mb = peak as f64 / 1024.0 / 1024.0,
            "Memory usage"
        );
    }
}
```

## 13. 常见问题排查手册

### 13.1 编译问题

**Q: tree-sitter 编译失败？**
```bash
# 清理后重新编译
cargo clean && cargo build --features core-languages

# 确保 tree-sitter CLI 已安装
cargo install tree-sitter-cli
```

**Q: 依赖下载失败？**
```bash
# 使用国内镜像源
export CARGO_REGISTRY=https://rsproxy.cn/crates.io-index
cargo build
```

### 13.2 运行时问题

**Q: Ollama 连接失败？**
```bash
# 检查 Ollama 是否运行
ollama list

# 手动启动 Ollama
ollama serve
```

**Q: 权限不足？**
```bash
# Linux/macOS
chmod +x target/release/agenthub

# Windows (PowerShell)
Set-ExecutionPolicy -Scope Process -ExecutionPolicy Bypass
```

### 13.3 性能问题

**Q: 首次响应很慢？**
- 这是正常的，LLM 需要加载模型到内存
- 后续请求会快很多（得益于缓存）
- 可以考虑预热：启动时发送一个空请求

**Q: 内存占用过高？**
- 检查是否启用了不必要的语言解析器
- 减小缓存大小（config 中调整 cache.max_size）
- 使用更小的模型（qwen:0.5b 用于开发测试）
